CREATE TABLE [dbo].[OperationResult] (
    [Id]           BIGINT        IDENTITY (1, 1) NOT NULL,
    [OperationId]  BIGINT        NOT NULL,
	[UserId]       BIGINT        NOT NULL,
	[Args]         NVARCHAR(1000)  NULL,
    [Result]       FLOAT        NULL,
    [CreationDate] DATETIME   NOT NULL,
    [Status]       INT NULL,
    [ExecTime]     BIGINT   NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC), 
    CONSTRAINT [FK_OperationResult_User] FOREIGN KEY ([UserId]) REFERENCES [Users]([Id]), 
    CONSTRAINT [FK_OperationResult_Operation] FOREIGN KEY ([OperationId]) REFERENCES [Operation]([Id])
);

CREATE TABLE [dbo].[Operation] (
    [Id]       BIGINT        IDENTITY (1, 1) NOT NULL,
    [Name]     NVARCHAR(1000) NOT NULL,
    [Rating]   INT NOT NULL,
    [ArgCount] INT        NOT NULL,
    [Owner]    NVARCHAR(1000) NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);
CREATE TABLE [dbo].[Users] (
    [Id]       BIGINT           IDENTITY (1, 1) NOT NULL,
    [Login]    NVARCHAR (50) NOT NULL,
    [Password] NVARCHAR(1000)           NOT NULL,
    [Roles]    NVARCHAR (10) NOT NULL,
    [Email]    NVARCHAR (50) NULL,
    [BirthDay] DATETIME          NULL,
    [Status]   INT NOT NULL,
    [Company]  NVARCHAR (1000) NULL,
    [Sex]      BIT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);
